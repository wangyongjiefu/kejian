1 请求后台接口list 获取 数据展示到页面上

  • 发送ajax请求
  • 在mounted 钩子中将数据渲染到页面上
axios.defaults.baseURL = 'http://localhost:3001/';

new Vue({
     data: {
                currentEditing: null,
                todos: []
     },    
      mounted() {
                  # 1.2 在mounted  钩子中将数据渲染到页面上
                this.getList()
            },
      methods: {
                  # 1.1 发送ajax请求
                getList() {
                    axios.get('list').then(res => {
                        console.log(res.data)
                        this.todos = res.data
                    })
                }
       }          

})
